<!-- Checkbox -->


<script>
var sHiddenCampoX='P2PedCampoX';
var aTextCampoX=new Array('8h às 10h','10h às 12h','12h às 14h','14h às 16h','16h às 18h');
var aValueCampoX=new Array('8h às 10h','10h às 12h','12h às 14h','14h às 16h','16h às 18h');
var sErrMsgCampoX='Selecione horário(s) preferido(s) de entrega.';

document.write("<input type=hidden name="+sHiddenCampoX+" id="+sHiddenCampoX+" onpropertychange=PedAlteraCampoX(this.value)>");

for(var i=0;i<aTextCampoX.length;i++)
  document.write("<input type=checkbox name=x"+sHiddenCampoX+" id=x"+sHiddenCampoX+"_"+i+" value='"+aValueCampoX[i]+"' onclick=AtualizaHiddenCampoX()><label for=x"+sHiddenCampoX+"_"+i+">"+aTextCampoX[i]+"</label><br>");


function PedAlteraCampoXFF(){PedAlteraCampoX(document.Form1.P2PedCampoX.value);}
if(navigator.appName.indexOf("Explorer")<0)document.getElementById(sHiddenCampoX).addEventListener('DOMAttrModified',PedAlteraCampoXFF,false);

PedAlteraCampoX("<ValorCampo>");

function PedAlteraCampoX(sValor){
  var aValores=sValor.split(", ");
  var iSelect=0,Achei,iArray;
  while(iSelect<document.Form1.xP2PedCampoX.length){
    Achei=false;
    iArray=0;
    while(iArray<aValores.length && !Achei){
      if(document.Form1.xP2PedCampoX[iSelect].value==aValores[iArray]){document.Form1.xP2PedCampoX[iSelect].checked=true;Achei=true;}
      else iArray++;
    }
    if(!Achei)document.Form1.xP2PedCampoX[iSelect].checked=false;
    iSelect++;
  }
}

function AtualizaHiddenCampoX(){
  var iSelect=0,sHiddenValor="";
  while(iSelect<document.Form1.xP2PedCampoX.length){
    if(document.Form1.xP2PedCampoX[iSelect].checked){
      if(sHiddenValor!="")sHiddenValor+=", ";
      sHiddenValor+=document.Form1.xP2PedCampoX[iSelect].value;
    }
    iSelect++;
  }
  document.Form1.P2PedCampoX.value=sHiddenValor;
}

function PedValidaCampoX(){
  var iSelect=0,Achei=false;
  while(iSelect<document.Form1.xP2PedCampoX.length && !Achei){
    if(document.Form1.xP2PedCampoX[iSelect].checked)Achei=true;
    else iSelect++;
  }
  if(!Achei){alert(sErrMsgCampoX);document.Form1.xP2PedCampoX[0].focus();return false;}
  else return true;
}
</script>
