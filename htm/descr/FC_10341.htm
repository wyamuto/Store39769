function ReadCookie(cookieName) {
 var theCookie=""+document.cookie;
 var ind=theCookie.indexOf(cookieName);
 if (ind==-1 || cookieName=="") return ""; 
 var ind1=theCookie.indexOf(';',ind);
 if (ind1==-1) ind1=theCookie.length; 
 return unescape(theCookie.substring(ind+cookieName.length+1,ind1));
}

Obs¹: Caso queira criar Flash personalizado para exibição da cesta, utilize o seguinte caminho do XML:

/sistema/XMLCart.asp?IDLoja=999999

Segue exemplo em ActionScript 2

xmlData.load("/sistema/XMLCart.asp?IDLoja=9999");

Segue exemplo em ActionScript 3

var xmlLoader:URLLoader = new URLLoader();
var xmlData:XML = new XML();
xmlLoader.addEventListener(Event.COMPLETE, LoadXML);
xmlLoader.load(new URLRequest("/sistema/XMLCart.asp?IDLoja=9999"));
