Segue abaixo exemplo de código HTML que pode ser utilizado em páginas personalizadas da loja (Ex: BarraTopo.htm, BarraLateral.htm, etc) para que o cliente possa cadastrar o e-mail na lista de assinantes de e-mail marketing da loja:

<table width=100% class=EstTabPedido align=center border=0>
<form name=FormCadNewsletter method=GET action="Newsletter.asp" onsubmit="return ValidaCadNewsletter(this);">
  <input type=hidden name=IDLoja value="<IDLojaNum>">
  <input type=hidden name=Y value="<Y>">
  <input type=hidden name=FormatoHTML value=True>
  <input type=hidden name=EnviarAssinante value="Enviar">
 <tr><td class=EstNomeCampo>Nome:</td></tr> 
 <tr><td><input type=text name=NomeAssinante size=10 maxlength=35 class=InputText style="width:100px;"></td></tr>
 <tr><td class=EstNomeCampo>Email:</td></tr> 
 <tr><td><input type=text name=Email size=10 maxlength=35 class=InputText style="width:100px;"></td></tr>
 <tr><td colspan=2 align=center><input type=submit class=InputButton name=ok value="Enviar"></td></tr>
</form>
</table>

<script>
function ValidaCadNewsletter(oForm) {
  function ValidaEmailCadNews(sEmail) {
    var regex=/^[a-zA-Z0-9._-]+@([a-zA-Z0-9.-]+\.)+[a-zA-Z0-9.-]{2,4}$/;
    return regex.test(sEmail);
  }
  if (oForm.NomeAssinante.value.length==""){
    alert("Campo 'Nome' não pode estar em branco!");
    oForm.NomeAssinante.focus();
    return false;
  }
  if (oForm.Email.value.length==""){
    alert("Campo 'E-mail' não pode estar em branco!");
    oForm.Email.focus();
    return false;
  }
  if (!ValidaEmailCadNews(oForm.Email.value)) {
    alert("Campo 'E-mail' inválido!");
    oForm.Email.focus();
    return false;
  }
}
</script>
