Para dar maior visibilidade ao botão continuar na página de pedido confirmado, posicione o botão Continuar no topo da página. Para fazer isto, adicione o código abaixo no arquivo ConfirmacaoPedido.htm. Se este arquivo não existir na pasta HTM crie um novo e envie para a pasta.

 <style>
#idTxtPedConfirmadoFC{height:30px;padding:10px;background-color:#007dd4;color:#fcfcfc;font-size:14pt;}
#idDadosPagtoFC{font-size:9pt;}
#EstNomePagto{color:#555555;font-size:11pt;font-weight:bold;}
.idTxtPrintFC{display:none;}
.EstPassosCompra{display:none;}
</style>

<script>
var iFormaPagto="<IDFormaPagamento>";

if(iFormaPagto=="3"){ //Cartão de crédito
  var sFormaPagto="<FormaPagamento> (<TipoCartao>)"
  setTimeout("document.getElementById('idDadosPagtoTitFC').innerHTML='<td class=EstTabTopo colspan=2 align=center>Forma de pagamento</td>';",1); //Altera titulo da parte inferior dos dados para pagamento
  }
  else{
  var sFormaPagto="<FormaPagamento>"
  setTimeout("document.getElementById('idDadosPagtoTitFC').style.display='none';",1); //Altera titulo da parte inferior dos dados para pagamento
}

var sDadosPagto="<br><br><table width=97% align=center class=EstTabPedido><tr height=25><td class=EstTabPedidoTit align=center>Dados para pagamento</td></tr><tr><td align=center><br><span id=EstNomePagto>"+ sFormaPagto +"</span><br><br>"+document.getElementById('idDadosPagtoFC').innerHTML+"</td></tr></table>"; //Instruções e botão para pagamento
var sBannerEbit="<br>";  //Banner do e-bit

setTimeout("document.getElementById('idTopoConfirmacaoFC').innerHTML=sDadosPagto+sBannerEbit+'<br><br><span class=Titulos>Informações do seu pedido:</span>';",1);  //Insere no topo da página de pedido confirmado as instruções para pagamento + banner do ebit
setTimeout("document.getElementById('idDadosPagtoFC').innerHTML='';",1); //Remove da parte inferior os dados para pagamento
</script>
