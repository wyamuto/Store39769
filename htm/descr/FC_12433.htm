Segue o passo-a-passo:


1º Passo:

Crie um novo widget e preencha o campo ID da posição com o nome Widget1.

Altere os campos abaixo StoreID, Partner e PosID de acordo com o ID da loja, o nome do parceiro cadastrado na loja e a posição na página onde o widget será exibido.

<script type='text/javascript'>
var _FCw_aProds=new Array();

var _FCw_prod=new Array();
_FCw_prod['StoreID']=999999;
_FCw_prod['Partner']='widget';
_FCw_prod['AdType']=1;
_FCw_prod['PosID']='Widget1';

_FCw_aProds[_FCw_aProds.length]=_FCw_prod;
(function(){
var fcw=document.createElement('script');
fcw.type='text/javascript';
fcw.async=true;
fcw.src=('https:'==document.location.protocol?'https':'http')+'://www.rumo.com.br/ProdWidget.js';
var s=document.getElementsByTagName('script')[0];
s.parentNode.insertBefore(fcw,s);
})();
</script>

2º Passo:

Escolha um novo modelo de widget e preencha o campo ID da posição com o nome Widget2. Insira os novos parâmetros do segundo modelo de widget e altere o parâmetro StoreID, Partner e PosID, conforme o 1º passo.

<script type='text/javascript'>
var _FCw_aProds=new Array();

//Widget modelo 1
var _FCw_prod=new Array();
_FCw_prod['StoreID']=999999;
_FCw_prod['Partner']='widget';
_FCw_prod['AdType']=1;
_FCw_prod['PosID']='Widget1';

//Widget modelo 2
var _FCw_prod=new Array();
_FCw_prod['StoreID']=999999;
_FCw_prod['Partner']='widget';
_FCw_prod['AdType']=2;
_FCw_prod['PosID']='Widget2';

_FCw_aProds[_FCw_aProds.length]=_FCw_prod;
(function(){
var fcw=document.createElement('script');
fcw.type='text/javascript';
fcw.async=true;
fcw.src=('https:'==document.location.protocol?'https':'http')+'://www.rumo.com.br/ProdWidget.js';
var s=document.getElementsByTagName('script')[0];
s.parentNode.insertBefore(fcw,s);
})();
</script>

3º Passo:

Insira as DIVs com os IDs abaixo nas posições onde deseja exibir os widgets.

<div id='Widget1'></div>
<div id='Widget2'></div>

OBS: Uma loja pode exibir widgets de outra loja. Entretanto, ela não poderá exibir widgets com seus próprios produtos.
