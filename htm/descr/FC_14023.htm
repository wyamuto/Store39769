<script>
function ValidaCampoLog(campo,msg,minimo){
 if(campo.value=='')
  {
  alert('Informe '+msg+'.');return false;
  }
 else
  {
   if(campo.value.length<minimo)
    {
    alert('Preencha '+msg+' corretamente.');return false;
    }
  }
  return true;
}

function Valida(oForm){
  if(!ValidaCampoLog(oForm.SenhaAtual,'a senha',6)){oForm.SenhaAtual.focus();return void(0);}
  if(!ValidaCampoLog(oForm.Email,'seu e-mail',10)){oForm.Email.focus();return void(0);}
  oForm.submit();
}
</script>
