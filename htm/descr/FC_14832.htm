2º Passo

Inclua o script abaixo ao final do seu arquivo LojaLib.js:

function fnCNPJInscrText(){
  var oTextList = document.querySelectorAll("#idTableRGFC .smText");
  if(oTextList && oTextList.length > 0){
    for(var i=0; i< oTextList.length; i++){
    var oText = oTextList[i];
    var sText = oText.textContent;
      if( (sText.toUpperCase()).indexOf("/MUNICIPAL") >= 0 ){
      oText.id="idTxtIncrCNPJ_FC";
      if(FC$.Page=="Register"){
          oText.innerHTML = "  Inscrição Estadual ou informe isento.";
      }else if(FC$.Page=="Checkout"){
          oText.innerHTML = "  Estadual ou informe isento";
    }
   }
  }
 }
  if(!document.querySelector("#idTableRGFC #idTxtIncrCNPJ_FC")){
  setTimeout(function(){fnCNPJInscrText();},500);
  }
}

3º Passo

Inclua no início do arquivo Rodape.htm o código abaixo:

<script>
if(FC$.Page=="Register" && FC$.Step=="Registration")fnCNPJInscrText();
</script>

Obs: Caso a loja utilize o Xtreme Checkout é necessário incluir a chamada da função no arquivo PedRodape.htm para que funcione corretamente no fechamento do pedido.
