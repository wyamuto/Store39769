<!-- BEGIN Google Avaliações do Consumidor  -->
<script>
Date.prototype.addDays = function(days) {
  this.setDate(this.getDate() + parseInt(days));
  return this;
};

function fnDateGoogle(){
  var currentDate=new Date();
  currentDate.addDays(7);
  var month=currentDate.getUTCMonth()+1;
  var day=currentDate.getUTCDate();
  var year=currentDate.getUTCFullYear();
  day=day+"";
  month=month+"";
  if(day.length==1)day="0"+day;
  if(month.length==1)month="0"+month;
  var newdateGoogle=year +"-"+ month +"-"+ day;
  return newdateGoogle; 
}

F$.fnLoadScript('https://apis.google.com/js/platform.js?onload=renderOptIn',true);
</script>

<script>
  window.renderOptIn = function() {
    window.gapi.load('surveyoptin', function() {
      window.gapi.surveyoptin.render(
        {
          "merchant_id": 9999999999,
          "order_id": "<NumPedido>",
          "email": "<EmailPedido>",
          "delivery_country": "BR",
          "estimated_delivery_date": fnDateGoogle()
        });
    });
  }
</script>
