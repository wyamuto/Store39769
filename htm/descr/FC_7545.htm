StatusTexto
NovoRecebido pela loja
CanceladoCancelado
Em aprovaçãoAguardando aprovação
PendenteCom pendências
AprovadoPagamento verificado
LiberadoLiberado para envio
RemetidoRemetido


É possível personalizar o texto exibido em cada status do pedido.


1º Passo

Altere o texto para cada status na função abaixo e inclua no arquivo LojaLib.js:

function AlteraTextoSitPedido(){
   var sitNovo=document.getElementById("idTxtSit1OrderFC");
   if(sitNovo)sitNovo.innerHTML="Status pedido Novo";
   else{  
     var sitCancelado=document.getElementById("idTxtSit2OrderFC");
     if(sitCancelado)sitCancelado.innerHTML="Status pedido Cancelado";
     else{  
      var sitEmAprovacao=document.getElementById("idTxtSit3OrderFC");
      if(sitEmAprovacao)sitEmAprovacao.innerHTML="Status pedido Em aprovação";
       else{    
         var sitPendente=document.getElementById("idTxtSit4OrderFC");
         if(sitPendente)sitPendente.innerHTML="Status pedido Pendente";
         else{    
           var sitAprovado=document.getElementById("idTxtSit5OrderFC");
           if(sitAprovado)sitAprovado.innerHTML="Status pedido Aprovado";
           else{      
            var sitLiberado=document.getElementById("idTxtSit6OrderFC");
            if(sitLiberado)sitLiberado.innerHTML="Status pedido Liberado";
             else{       
              var sitRemetido=document.getElementById("idTxtSit7OrderFC");
              if(sitRemetido)sitRemetido.innerHTML="Status pedido Remetido";
             }
           }
         }
       }
     }
   }
}

2º Passo

inclua no arquivo Rodape.htm o script abaixo:

<script>
if(document.getElementById("idOrdersTrackFC"))AlteraTextoSitPedido();
</script>
