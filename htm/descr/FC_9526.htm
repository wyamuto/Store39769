Segue abaixo o procedimento de implantação deste recurso:


1º passo

Cadastre o campo adicional, clicando na seta ao lado direito do campo na página Dados da loja do site administrativo.

Na janela Campos adicionais de clientes, informe Período de entrega no campo Campo 1 da tabela Entrega do pedido e selecione no campo 30 no campo Máx.


2º passo (Checkout Passo-a-passo)

Crie um arquivo chamado PedCampo1.htm com o conteúdo abaixo e envie este arquivo para a pasta htm da loja.


<script>var sHiddenCampo1='P2PedCampo1';

var aValueCampo1=new Array('Sem preferência','Manhã (das 7h as 12h)','Tarde (12h as 18h)','Noite (18h as 23h)');
var sErrMsgCampo1='Selecione período preferencial de entrega.';

document.write("<input type=hidden name="+sHiddenCampo1+" id="+sHiddenCampo1+" onpropertychange=PedAlteraCampo1(this.value)>");

  document.write("<input type=radio name=x"+sHiddenCampo1+" id=x"+sHiddenCampo1+"_"+i+" value='"+aValueCampo1[i]+"' onclick=document.Form1."+sHiddenCampo1+".value=this.value><label for=x"+sHiddenCampo1+"_"+i+">"+aTextCampo1[i]+"</label><br>");


function PedAlteraCampo1FF(){PedAlteraCampo1(document.Form1.P2PedCampo1.value);}

function PedAlteraCampo1(sValor){
  var iSelect=0,Achei=false;
  while(iSelect<document.Form1.xP2PedCampo1.length && !Achei){
    if(document.Form1.xP2PedCampo1[iSelect].value==sValor){document.Form1.xP2PedCampo1[iSelect].checked=true;Achei=true;}
    else iSelect++;
  }
}

function PedValidaCampo1(){
  var iSelect=0,Achei=false;
  while(iSelect<document.Form1.xP2PedCampo1.length && !Achei){
    if(document.Form1.xP2PedCampo1[iSelect].checked)Achei=true;
    else iSelect++;
  }
  if(!Achei){alert(sErrMsgCampo1);document.Form1.xP2PedCampo1[0].focus();return false;}
  else return true;
}
</script>
