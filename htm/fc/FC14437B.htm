<!-- começo :: zoom com várias imagens -->
<script>
var imgDetAll = "<ImagemProdDet>";
var imgAmpAll = "<ImagemProdAmp>";
var novoArray = imgDetAll.split(',');
var novoArrayAmp = imgAmpAll.split(',');
var CountImgDet=novoArray.length;
var CountImgAmp=novoArrayAmp.length;

if (imgDetAll=="" || imgAmpAll==""){
  document.write("");
}
else{
  if (CountImgDet==CountImgAmp){
    var FC_MaxImages=CountImgDet-1;
    MostraZoomImgFC();
  }
  else{
    var FC_MaxImages=0;
    MostraZoomImgFC();
  }
}

function MostraZoomImgFC(){
  for (var i=0;i<=FC_MaxImages;i++)
  {
    if(i==0)
    {
       imgDetMini=novoArray[i];
       imgAmpMini=novoArrayAmp[i];
       document.write("<a href="+imgAmpMini+" title=<NomeProd> class=MagicZoomPlus id=zoom2 rel='selectors-class:active; zoom-width:350px; zoom-height:350px; selectors-change:mouseover;'><img src="+ imgDetMini +"></a><br>");
       document.write('<p><a onclick="MagicZoomPlus.expand(zoom2); return false;" href="#">Ampliar imagem</a></p>');
       document.write("<a href="+imgAmpMini+"  rel='zoom-id:zoom2;' rev="+ imgDetMini +"><img src="+ imgDetMini +" width=75 height=45 class=ZoomIMG2></a>");
    }
    else{
          imgDetMini="<BaseLoja>prod/"+novoArray[i];
          imgAmpMini="<BaseLoja>prod/"+novoArrayAmp[i];
          document.write("<a href="+imgAmpMini+" rel='zoom-id:zoom2;' rev="+ imgDetMini +"><img src="+ imgDetMini +" width=75 height=45 class=ZoomIMG2></a>");
    }
  }
}
</script>
<!-- fim :: zoom com várias imagens -->
